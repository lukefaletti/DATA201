In [5]:
data = data.astype({"Currency": str, "Date": str, "Open": float, "High": float, "Low": float, "Close": float})
In [6]:
data["Date"] = pd.to_datetime(data["Date"])
data["Volume"] = data["Volume"].str.replace(',','')
data["Volume"] = data["Volume"].replace('-', np.nan)
data["Volume"] = pd.to_numeric(data["Volume"])
data["Market Cap"] = data["Market Cap"].str.replace(',','')
data["Market Cap"] = data["Market Cap"].replace('-', np.nan)
data["Market Cap"] = pd.to_numeric(data["Market Cap"])
